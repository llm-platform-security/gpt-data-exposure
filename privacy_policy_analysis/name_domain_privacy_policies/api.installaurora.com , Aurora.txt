        let lastScrollTop = 0;

            let st = window.pageYOffset || document.documentElement.scrollTop;

                // If scrolling down
                document.querySelector("header").classList.add("header-hide");
            } else {

                document.querySelector("header").classList.remove("header-hide");
            }
            lastScrollTop = st <= 0 ? 0 : st;
        }, false);
