        var handleArrowClick = function (id) {
            var contentTargets = document.getElementsByClassName('content-wrap')
            var arrowTargets = document.getElementsByClassName('arrow')
            faqList.forEach((item) => {
                if (id !== item.id) {
                    item.isShow = false;
                    contentTargets[item.id - 1].style.display = 'none'
                    arrowTargets[item.id - 1].classList = 'arrow rotate'
                } else {
                    item.isShow = !item.isShow
                    if (item.isShow) {
                        contentTargets[item.id - 1].style.display = 'block'
                        arrowTargets[item.id - 1].classList = 'arrow'
                    } else {
                        contentTargets[item.id - 1].style.display = 'none'
                        arrowTargets[item.id - 1].classList = 'arrow rotate'
                    }
                }
            });
        }

            }
