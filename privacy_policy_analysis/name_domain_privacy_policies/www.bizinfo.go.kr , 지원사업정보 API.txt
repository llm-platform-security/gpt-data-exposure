	function fn_searchUrl() {
    	var activeTabId = $(".menu_new_info .active").attr("id");

    	var dataType = $("#dataType option:selected").val();
    	var searchCnt = $("#searchCnt").val();

    	var searchSportsportRearmCode = "";

    	// 분야별 해시태그
    	$("input[name=searchSportsportRearmCode]:checked").each(function (i) {
    	    if ($("input[name=searchSportsportRearmCode]:checked").length > 0) {
    	        if (i == 0) {
    	            searchSportsportRearmCode += $(this).val();
    	        } else {
     	           searchSportsportRearmCode += "," + $(this).val();
     	       }
   	 	    }
    	});

    	// 지역별 해시태그
    	$("input[name=searchAreaCode]:checked").each(function (i) {
       		 if ($("input[name=searchAreaCode]:checked").length > 0) {
           		 if (i == 0) {
               	 if (searchSportsportRearmCode != "") {
                    searchSportsportRearmCode += "," + $(this).val();
               	 } else {
                    searchSportsportRearmCode += $(this).val();
                	}
            	} else {
               		searchSportsportRearmCode += "," + $(this).val();
           	 }
        	}
    	});

		var url = "&dataType=" + dataType;
    	var flag = false;

    	if (searchSportsportRearmCode != "") {
        	if (flag) {
        	    url += "?";
         	   flag = false;
        	} else {
        	    url += "&";
        	}
        	url += "hashtags=" + searchSportsportRearmCode;
    	}

    	if (searchCnt != "") {

        	if (flag) {
        	    url += "?";
        	    flag = false;
       		} else {
        	    url += "&";
        	}

       	 url += "searchCnt=" + searchCnt;

    	}
    	$("#createUrl").val(url);
	}

	function fn_setCode(obj, name) {

        var codeVal = obj.val();
        var checkedCnt = 0;
        var inputName = "";

        if (name == "sportsportRearm") {
            inputName = "searchSportsportRearm";
        }else {
            inputName = "searchArea";
        }

        if (!$("#chk_" + name + "_" + codeVal).is(":checked")) {
            $("#chk_" + name + "_" + codeVal).attr("checked", false);
            $("#la_" + name + "_" + codeVal).removeClass();
        } else {
            $("#chk_" + name + "_" + codeVal).attr("checked", true);
            $("#la_" + name + "_" + codeVal).addClass("check");
        }

        $("input[name=" + inputName + "Code]").each(function () {
            if ($(this).is(":checked")) {
                checkedCnt++;
            }
        });
    }

		StringBuilder urlBuilder = new StringBuilder("http://X.X.X.X:X/test?cftfcKey=XXXXX"); /*URL*/
		URL url = new URL(urlBuilder.toString());
		HttpURLConnection conn = (HttpURLConnection) url.openConnection();
		conn.setRequestMethod("GET");
		conn.setRequestProperty("Content-type", "application/json");
		System.out.println("Response code: " + conn.getResponseCode());
		BufferedReader rd;
		
		if (conn.getResponseCode()>=200 && conn.getResponseCode() <=300){
			rd = new BufferedReader(new InputStreamReader(conn.getInputStream()));
		} else {
			rd = new BufferedReader(new InputStreamReader(conn.getErrorStream()));
		}
		
		StringBuilder sb = new StringBuilder();
		String line;
		while ((line = rd.readLine()) != null) {
			sb.append(line);
		}
		rd.close();
		conn.disconnect();
		System.out.println(sb.toString());
	}
}
