<div class="devsite-article-body clearfix
  ">

  
    
  <p>
    Use the PageSpeed Insights API to:
  </p>
  <ul>
    <li>Measure the performance of a web page.</li>
    <li>Get suggestions on how to improve the page's performance, accessibility, and SEO.</li>
    <li>
      Learn how to convert the page into a <a href="/web/progressive-web-apps">Progressive
      Web App.</a>
    </li>
  </ul>
  
  
  
  
  
  
  <p>
    The PageSpeed Insights API returns real-world data from the <a href="https://developer.chrome.com/docs/crux/">Chrome User
    Experience Report</a> and lab data from <a href="/web/tools/lighthouse">Lighthouse</a>.
  </p>
  <aside class="note">
    We plan to discontinue including real-world data from the
    <a href="https://developer.chrome.com/docs/crux/"'>Chrome User Experience Report</a> in this API. We recommend the
    <a href="https://developer.chrome.com/docs/crux/api/">CrUX API</a> (<a href="https://developer.chrome.com/blog/chrome-ux-report-api/">guide</a>) or the
    <a href="https://developer.chrome.com/docs/crux/history-api/">CrUX History API</a>
    (<a href="https://developer.chrome.com/blog/chrome-ux-report-history-api/">guide</a>) instead.
  </aside>
  <h2 id="prerequisites" data-text="Prerequisites" tabindex="-1">Prerequisites</h2>
  <p>
    This tutorial assumes that you have general knowledge of command line interfaces and web
    development.
  </p>
  <h2 id="key" data-text="Optional: Set up an API key" tabindex="-1">Optional: Set up an API key</h2>
  
  <p>
    If you're just trying out the PageSpeed Insights API, you don't need an API key. If you
    plan on using the API in an automated way and making multiple queries per second, you'll
    need an API key.
  </p>
  
  
  
  
  <h3 id="APIKey" data-text="   Acquiring and using an API key " tabindex="-1">
  Acquiring and using an API key
</h3>




  <a tabindex="0" class="devsite-api-getstarted-widget gc-analytics-event button button-primary"
     data-category="Site-Wide Custom Events"
     data-label="Get Henhouse Key"
     data-api-id="pagespeedonline">Get a Key</a>
  <p>Or create one in the <a href="https://console.cloud.google.com/apis/credentials">Credentials page</a>.</p>


<p>After you have an API key, your application can append the query parameter
<code translate="no" dir="ltr">key=<var translate="no">yourAPIKey</var></code> to all request URLs.</p>

<p>The API key is safe for embedding in URLs; it doesn't need any encoding.</p>
  <h2 id="explorer" data-text="API Explorer" tabindex="-1">API Explorer</h2>
  
  <p>
    To make calls to the PageSpeed Insights API without writing any code, check out the
    <a href="/apis-explorer#p/pagespeedonline/v5/">API Explorer</a>.
  </p>

  <h2 id="javascript" data-text="JavaScript" tabindex="-1">JavaScript</h2>
  <p>To try out the PageSpeed Insights API from JavaScript:</p>
  <ol>
    <li>
      <p>
        Copy the code below. The code uses the PageSpeed Insights API to measure the performance of
        <code translate="no" dir="ltr">https://developers.google.com</code>, and then it displays the results.
      </p>
      <pre style="height:500px;"
           class="prettyprint notranslate devsite-click-to-copy
                  lang-html" dir="ltr">&lt;script&gt;
function run() {
  const url = setUpQuery();
  fetch(url)
    .then(response =&gt; response.json())
    .then(json =&gt; {
      // See https://developers.google.com/speed/docs/insights/v5/reference/pagespeedapi/runpagespeed#response
      // to learn more about each of the properties in the response object.
      showInitialContent(json.id);
      const cruxMetrics = {
        &quot;First Contentful Paint&quot;: json.loadingExperience.metrics.FIRST_CONTENTFUL_PAINT_MS.category,
        &quot;First Input Delay&quot;: json.loadingExperience.metrics.FIRST_INPUT_DELAY_MS.category
      };
      showCruxContent(cruxMetrics);
      const lighthouse = json.lighthouseResult;
      const lighthouseMetrics = {
        &#39;First Contentful Paint&#39;: lighthouse.audits[&#39;first-contentful-paint&#39;].displayValue,
        &#39;Speed Index&#39;: lighthouse.audits[&#39;speed-index&#39;].displayValue,
        &#39;Time To Interactive&#39;: lighthouse.audits[&#39;interactive&#39;].displayValue,
        &#39;First Meaningful Paint&#39;: lighthouse.audits[&#39;first-meaningful-paint&#39;].displayValue,
        &#39;First CPU Idle&#39;: lighthouse.audits[&#39;first-cpu-idle&#39;].displayValue,
        &#39;Estimated Input Latency&#39;: lighthouse.audits[&#39;estimated-input-latency&#39;].displayValue
      };
      showLighthouseContent(lighthouseMetrics);
    });
}

function setUpQuery() {
  const api = &#39;https://www.googleapis.com/pagespeedonline/v5/runPagespeed&#39;;
  const parameters = {
    url: encodeURIComponent(&#39;https://developers.google.com&#39;)
  };
  let query = `${api}?`;
  for (key in parameters) {
    query += `${key}=${parameters[key]}`;
  }
  return query;
}

function showInitialContent(id) {
  document.body.innerHTML = &#39;&#39;;
  const title = document.createElement(&#39;h1&#39;);
  title.textContent = &#39;PageSpeed Insights API Demo&#39;;
  document.body.appendChild(title);
  const page = document.createElement(&#39;p&#39;);
  page.textContent = `Page tested: ${id}`;
  document.body.appendChild(page);
}

function showCruxContent(cruxMetrics) {
  const cruxHeader = document.createElement(&#39;h2&#39;);
  cruxHeader.textContent = &quot;Chrome User Experience Report Results&quot;;
  document.body.appendChild(cruxHeader);
  for (key in cruxMetrics) {
    const p = document.createElement(&#39;p&#39;);
    p.textContent = `${key}: ${cruxMetrics[key]}`;
    document.body.appendChild(p);
  }
}

function showLighthouseContent(lighthouseMetrics) {
  const lighthouseHeader = document.createElement(&#39;h2&#39;);
  lighthouseHeader.textContent = &quot;Lighthouse Results&quot;;
  document.body.appendChild(lighthouseHeader);
  for (key in lighthouseMetrics) {
    const p = document.createElement(&#39;p&#39;);
    p.textContent = `${key}: ${lighthouseMetrics[key]}`;
    document.body.appendChild(p);
  }
}

run();
&lt;/script&gt;
</pre>
    </li>
