<script type="text/javascript">

  $(window).on("load", function () {
    const commonLogin = new CommonLogin();
    jQuery.fn.serializeObject = function () {
      var obj = null;
      try {
        if (this[0].tagName && this[0].tagName.toUpperCase() == "FORM") {
          var arr = this.serializeArray();
          if (arr) {
            obj = {};
            jQuery.each(arr, function () {
              obj[this.name] = this.value;
            });
          }
        }
      } catch (e) {
        alert(e.message);
      } finally {
      }
      return obj;
    }

    const DEFAULT_ERROR_MESSAGE = "문제가 발생하였습니다. 관리자에게 문의하여주세요.";
    // Meta에서 csrf 정보 가지고 오기
    const token = document.querySelector("meta[name='_csrf']").content;
    const header = document.querySelector("meta[name='_csrf_header']").content;

    let button = $("#login-btn")
    button.click((e) => {
      let form = document.getElementById("login-form");
      let formData = new FormData(form);
      let password = formData.get('password');
      formData.set('password', btoa(password));

      let object = {};
      formData.forEach(function(value, key){
        object[key] = value;
      });
      let json = JSON.stringify(object);

      $.ajax({
        cache: false,
        url: "/sso/login",
        type: "post",
        contentType: "application/json",
        data: json,
        beforeSend : function(xhr){
          xhr.setRequestHeader(header, token);
        }
      }).then((response) => {
        if(!response || !response.redirectUrl) {
          return alert(DEFAULT_ERROR_MESSAGE);
        }
        commonLogin.saveUsernameToCookie();
        const redirectUrl = response.redirectUrl;
        window.location.href = redirectUrl;
      }).catch((response) => {
        if(response && response.responseText) {
          return alert(response.responseText)
        }

        return alert(DEFAULT_ERROR_MESSAGE);
      })
    });
  })
</script>
