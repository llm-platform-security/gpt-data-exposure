    function customConfirm(message, btnYes, btnNo) {
        const overlayStyle = `
                position: fixed;
                top: 0;
                left: 0;
                width: 100%;
                height: 100%;
                background-color: rgba(0, 0, 0, 0.5); /* Semi-transparent black overlay */
                z-index: 99; /* Ensure the overlay is on top of everything */
            `;

        class DialogHandler {
            constructor() {
                this.lockContent();
                this.createDialog(message, btnYes, btnNo);
                this.addEventListeners();
            }

            lockContent() {
                this.overlay = document.createElement('div');
                this.overlay.style.cssText = overlayStyle;
                document.body.appendChild(this.overlay);
            }

            unlockContent() {
                if (this.overlay) {
                    document.body.removeChild(this.overlay);
                }
            }

            createDialog(message, btnYes, btnNo) {
                this.dialog = document.createElement('div');
                let buttonsHTML = '';
                if (btnYes) {
                    buttonsHTML += `<button id="yes-button" class="btn btn-primary">${btnYes}</button>`;
                }
                if (btnNo) {
                    buttonsHTML += `<button id="no-button" class="btn btn-light">${btnNo}</button>`;
                }
                this.dialog.innerHTML = `
                        <div class="custom-dialog">
                            <p>${message}</p>                            
                                <div class="text-center">${buttonsHTML}</div>                            
                        </div>
                    `;

                // Add some CSS styling to make it visible
                this.dialog.style.position = 'fixed';
                this.dialog.style.top = '50%';
                this.dialog.style.left = '50%';
                this.dialog.style.transform = 'translate(-50%, -50%)';
                this.dialog.style.backgroundColor = 'white';
                this.dialog.style.padding = '40px';
                this.dialog.style.borderRadius = '8px';
                this.dialog.style.zIndex = 101;
                this.dialog.style.boxShadow = '0px 0px 10px rgba(0, 0, 0, 0.5)';
                document.body.appendChild(this.dialog);
            }

            addEventListeners() {
                if (btnYes) {
                    document.getElementById('yes-button').addEventListener('click', () => {
                        document.body.removeChild(this.dialog);
                        this.unlockContent();
                        this.resolve(true);
                    });
                }
                if (btnNo) {
                    document.getElementById('no-button').addEventListener('click', () => {
                        document.body.removeChild(this.dialog);
                        this.unlockContent();
                        this.resolve(false);
                    });
                }
            }

            show() {
                return new Promise((resolve, reject) => {
                    this.resolve = resolve;
                });
            }
        }

        return new DialogHandler().show();
    }
