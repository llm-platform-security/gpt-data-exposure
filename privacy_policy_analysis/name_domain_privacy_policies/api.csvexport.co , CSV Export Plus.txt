        <!-- Typing Animation Effect -->
        <script>
            var favicon = document.getElementById('favicon');
            var plusWordElement = document.getElementById('plus-word');
            var word = " Plus";
            var index = 0;
            var isDeleting = false;
            var pauseBetween = 2000;
            var timeoutId;

            function typePlusWord() {
                clearTimeout(timeoutId); // clear the previous timeout
                if (!isDeleting && index <= word.length) {
                    plusWordElement.innerHTML = `<span style='color: #ffec00;'>${word.slice(0, index)}</span>`;
                    favicon.href = 'images/favicon-2.png';
                    index++;
                    timeoutId = setTimeout(typePlusWord, 150);
                } else if (!isDeleting && index > word.length) {
                    isDeleting = true;
                    timeoutId = setTimeout(typePlusWord, pauseBetween);
                } else if (isDeleting && index >= 0) {
                    plusWordElement.innerHTML = `<span style='color: #ffec00;'>${word.slice(0, index)}</span>`;
                    favicon.href = (index > 0) ? 'images/favicon-1.png' : 'images/favicon-2.png';
                    index--;
                    timeoutId = setTimeout(typePlusWord, 150);
                } else if (isDeleting && index < 0) {
                    isDeleting = false;
                    timeoutId = setTimeout(typePlusWord, pauseBetween);
                }
            }

            timeoutId = setTimeout(typePlusWord, pauseBetween);
        </script>
